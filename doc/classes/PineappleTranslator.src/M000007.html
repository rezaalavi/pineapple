  def initialize(current_step)
    @script= ""

    steps_root_path = ENV['PINEAPPLE_STEPS_PATH']
    if steps_root_path.nil? || steps_root_path.empty?
      steps_root_path = "#{Rails.root}/pineapple/steps/"
    end
    
    @file_name = "#{steps_root_path}#{current_step}"
    @step_body = load_step
  end
